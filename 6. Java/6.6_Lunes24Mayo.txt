package lunes24mayo;
import java.util.Scanner;

public class Lunes24Mayo {

   
    public static void main(String[] args) {
        // La calculadora.
        Scanner leer=new Scanner (System.in);

String seguir = "S"; 
        String op ;
        int num1, num2, resul=0;
while ( "S".equals(seguir)) {
    
    System.out.println("Ingresa dos números para empezar ");
        num1=leer.nextInt();
         num2=leer.nextInt();    
         
         System.out.println("Qué quieres hacer con ellos? \n+\n-\n/\n* ");
        op=leer.next();
         switch (op){
             case "+":
                    System.out.println("Seleccionaste suma");
                    resul=num1+num2; 
                 
                    break;
             case "-":
                    System.out.println("Seleccionaste resta");
                    resul=num1-num2; 
                   
                    break;
             case "/":
                    System.out.println("Seleccionaste dividir");
                    resul=num1/num2; 
                 
                    break;
             case "*":
                    System.out.println("Seleccionaste multiplicación");
                    resul=num1*num2; 
                  
                    break;       
                   
        } System.out.println("tu resultado es " +resul);  
     
        System.out.println("Quieres hacer otra operación? S/N");
        
        seguir=leer.next();  
}
    
    }
}
