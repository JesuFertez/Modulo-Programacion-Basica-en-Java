package lunes7junioherencia;


public class Electrodomestico {
private   int voltaje;
private   String ClasificacionEnergetica;
private   String gama;
private String marca;
private String color;

   public Electrodomestico(){};
   
   public Electrodomestico(int voltaje,String ClasificacionEnergetica,String gama,String marca,String color){
   this.voltaje=voltaje;
   this.ClasificacionEnergetica=ClasificacionEnergetica;
   this.gama=gama;
   this.color=color;
   this.marca=marca;
   }
   //Setter y Getter
   public void setvoltaje(){
   this.voltaje=voltaje;
   }
   public int getvoltaje(){
       return this.voltaje;
   }
   public void setClasificacionEnergetica(){
   this.ClasificacionEnergetica=ClasificacionEnergetica;
   }
   public String getClasificacionEnergetica(){
       return this.ClasificacionEnergetica;
   }
   public void setgama(){
   this.gama=gama;
   }
   public String getgama(){
       return this.gama;
   }
   
   public void setmarca(){
   this.marca=marca;
   }
    public String getmarca(){
       return this.marca;
    }
    public void setcolor(){
   this.color=color;
   }
    public String getcolor(){
       return this.color;
    }   
    
    
 //Métodos
   public void mostrarDatos(){
       System.out.println("El voltaje es "+voltaje+"La Clasificación Energética es "+ ClasificacionEnergetica+"La Gama "+gama+"La Marca es "+ marca+"El color es "+color);
   }
}
